{
    use InteractsWithSockets, SerializesModels;

    public function __construct(
        public Service $service
    ) {}

    public function broadcastOn(): Channel
    {
        return new Channel('services.' . $this->service->id);
    }

    public function broadcastWith(): array
    {
        return [
            'service_id' => $this->service->id,
            'title' => $this->service->title,
            'price' => $this->service->price,
            'updated_at' => $this->service->updated_at->toISOString(),
        ];
    }
}

Notification Service
php// app/Domains/Users/Services/NotificationService.php
<?php

namespace App\Domains\Users\Services;

use App\Domains\Users\Models\User;
use App\Domains\Users\Models\NotificationPreference;
use Illuminate\Support\Facades\Notification;

class NotificationService
{
    public function setPreference(User $user, string $type, string $channel, bool $enabled): void
    {
        NotificationPreference::updateOrCreate(
            [
                'user_id' => $user->id,
                'notification_type' => $type,
                'channel' => $channel,
            ],
            ['enabled' => $enabled]
        );
    }

    public function getPreferences(User $user): array
    {
        $preferences = $user->notificationPreferences()
            ->get()
            ->groupBy('notification_type');

        $result = [];
        foreach (NotificationPreference::TYPES as $type => $label) {
            $result[$type] = [
                'label' => $label,
                'channels' => [],
            ];
            
            foreach (NotificationPreference::CHANNELS as $channel) {
                $pref = $preferences->get($type)?->firstWhere('channel', $channel);
                $result[$type]['channels'][$channel] = $pref ? $pref->enabled : true;
            }
        }

        return $result;
    }

    public function sendNotification(User $user, $notification, string $type): void
    {
        $channels = [];
        
        foreach (NotificationPreference::CHANNELS as $channel) {
            if ($user->shouldReceiveNotification($type, $channel)) {
                $channels[] = $channel;
            }
        }

        if (!empty($channels)) {
            Notification::send($user, $notification);
        }
    }
}

public function updateNotifications(Request $request)
{
    $validated = $request->validate([
        'preferences' => 'required|array',
        'preferences.*' => 'array',
        'preferences.*.*' => 'boolean',
    ]);

    $notificationService = app(NotificationService::class);
    $user = $request->user();

    foreach ($validated['preferences'] as $type => $channels) {
        foreach ($channels as $channel => $enabled) {
            $notificationService->setPreference($user, $type, $channel, $enabled);
        }
    }

    return back()->with('status', 'Notification preferences updated');
}

php// app/Domains/Listings/Services/ListingImageService.php
<?php

namespace App\Domains\Listings\Services;

use App\Domains\Listings\Models\ListingImage;
use App\Domains\Common\Contracts\StorageServiceInterface;
use Illuminate\Database\Eloquent\Model;
use Illuminate\Http\UploadedFile;
use Intervention\Image\Facades\Image;

class ListingImageService
{
    public function __construct(
        private StorageServiceInterface $storage
    ) {}

    public function attachToModel(Model $model, UploadedFile $file, array $options = []): ListingImage
    {
        // Validate file
        $this->validateImage($file);

        // Generate unique filename
        $filename = time() . '_' . uniqid() . '.' . $file->getClientOriginalExtension();
        $path = 'uploads/listings/' . $filename;

        // Store original
        $storedPath = $this->storage->store($file, 'uploads/listings');

        // Generate thumbnail
        $thumbnailPath = $this->generateThumbnail($file, $storedPath);

        // Create database record
        return $model->images()->create([
            'path' => $storedPath,
            'thumbnail_path' => $thumbnailPath,
            'alt_text' => $options['alt_text'] ?? $file->getClientOriginalName(),
            'order_index' => $options['order_index'] ?? 0,
            'is_primary' => $options['is_primary'] ?? false,
        ]);
    }

    public function deleteImage(ListingImage $image): bool
    {
        // Delete files from storage
        $this->storage->delete($image->path);
        if ($image->thumbnail_path) {
            $this->storage->delete($image->thumbnail_path);
        }

        // Delete database record
        return $image->delete();
    }

    private function validateImage(UploadedFile $file): void
    {
        $file->validate([

            $thumbnailFilename = 'thumb_' . basename($originalPath);
            $thumbnailPath = 'uploads/listings/thumbnails/' . $thumbnailFilename;

            // Load image
            $img = Image::make($file);

            // Resize to 300x300 (fit mode)
            $img->fit(300, 300, function ($constraint) {
                $constraint->upsize();
            });

            // Save thumbnail
            $img->save(storage_path('app/public/' . $thumbnailPath), 80);

            return $thumbnailPath;
        } catch (\Exception $e) {
            \Log::error('Thumbnail generation failed: ' . $e->getMessage());
            return null;
        }
    }

public function render($request, Throwable $exception)
{
    // Domain exceptions
    if ($exception instanceof DomainException) {
        if ($request->expectsJson()) {
            return response()->json([
                'message' => $exception->getMessage(),
                'errors' => $exception instanceof ValidationException 
                    ? $exception->getErrors() 
                    : [],
            ], $exception->getStatus());
        }
        
        return back()
            ->withErrors(['error' => $exception->getMessage()])
            ->withInput();
    }

    // Laravel validation
    if ($exception instanceof \Illuminate\Validation\ValidationException) {
        if ($request->expectsJson()) {
            return response()->json([
                'message' => 'Validation failed',
                'errors' => $exception->errors(),
            ], 422);
        }
    }

    return parent::render($request, $exception);

      newFiles.forEach(file => {

          preview: URL.createObjectURL(file)
        });
      });

      this.files.splice(index, 1);
    },

    async upload() {
      this.uploading = true;
      const formData = new FormData();
      
      this.files.forEach((f, i) => {
        formData.append(`files[${i}]`, f.file);
      });

      try {
        const response = await fetch('/api/things/{{ $thingId }}/media/bulk', {
          method: 'POST',
          body: formData,
          headers: {
            'X-CSRF-TOKEN': document.querySelector('meta[name="csrf-token"]').content
          }
        });

        if (response.ok) {
          this.files = [];
          location.reload(); // Or update state
        }
      } catch (error) {
        console.error(error);
      } finally {
        this.uploading = false;
      }
    }

<script>
function toastManager() {
  return {
    toasts: [],
    nextId: 1,

    addToast(data) {
      const toast = {
        id: this.nextId++,
        type: data.type || 'info',
        title: data.title || '',
        message: data.message || '',
        visible: true,
      };

      this.toasts.push(toast);

      // Auto-remove after 5 seconds
      setTimeout(() => {
        this.removeToast(toast.id);
      }, 5000);
    },

    removeToast(id) {
      const toast = this.toasts.find(t => t.id === id);
      if (toast) {
        toast.visible = false;
        // Remove from array after animation
        setTimeout(() => {
          this.toasts = this.toasts.filter(t => t.id !== id);
        }, 300);
      }
    }
  }
}

javascript// Modal focus trap
function modalComponent() {
  return {
    open: false,
    
    openModal() {
      this.open = true;
      this.$nextTick(() => {
        // Focus first focusable element
        this.$refs.modalContent.querySelector('button, input, select, textarea, a[href]')?.focus();
      });
    },
    
    closeModal() {
      this.open = false;
      // Return focus to trigger element
      this.$refs.trigger?.focus();
    },
    
    // Trap focus within modal
    handleTab(e) {
      const focusableElements = this.$refs.modalContent.querySelectorAll(
        'button, [href], input, select, textarea, [tabindex]:not([tabindex="-1"])'
      );
      const firstElement = focusableElements[0];
      const lastElement = focusableElements[focusableElements.length - 1];

      if (e.shiftKey && document.activeElement === firstElement) {
        lastElement.focus();
        e.preventDefault();
      } else if (!e.shiftKey && document.activeElement === lastElement) {
        firstElement.focus();
        e.preventDefault();
      }
    }
  }
}

      } finally {
        this.submitting = false;

javascript// Debounce search input (reduce API calls)
function debounce(func, wait) {
    let timeout;
    return function executedFunction(...args) {
        const later = () => {
            clearTimeout(timeout);
            func(...args);
        };
        clearTimeout(timeout);
        timeout = setTimeout(later, wait);
    };
}

// Usage in Alpine component
{
    searchQuery: '',
    
    init() {
        this.debouncedSearch = debounce(this.performSearch, 300);
        this.$watch('searchQuery', () => this.debouncedSearch());
    },
    
    performSearch() {
        fetch(`/api/search?q=${this.searchQuery}`)
            .then(r => r.json())
            .then(data => this.results = data);
    }
}
